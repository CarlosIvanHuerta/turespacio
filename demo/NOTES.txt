// models/Database.php
<?php
class Database {
    private $host = 'localhost';
    private $db_name = 'tu_base';
    private $username = 'usuario';
    private $password = 'clave';
    public $conn;

    public function conectar() {
        $this->conn = null;
        try {
            $this->conn = new PDO("mysql:host=" . $this->host . ";dbname=" . $this->db_name,
                $this->username, $this->password);
            $this->conn->exec("set names utf8");
        } catch(PDOException $exception) {
            echo "Error de conexión: " . $exception->getMessage();
        }
        return $this->conn;
    }

    public function obtenerPosts() {
        $conn = $this->conectar();
        $stmt = $conn->prepare("SELECT id, titulo, resumen FROM posts LIMIT 3");
        $stmt->execute();
        return $stmt->fetchAll(PDO::FETCH_ASSOC);
    }

    public function obtenerComentarios($postId) {
        $conn = $this->conectar();
        $stmt = $conn->prepare("SELECT nombre, comentario FROM comentarios WHERE post_id = :post_id");
        $stmt->bindParam(':post_id', $postId, PDO::PARAM_INT);
        $stmt->execute();
        return $stmt->fetchAll(PDO::FETCH_ASSOC);
    }

    public function obtenerDatosGenerales() {
        return [
            'mensaje' => 'Bienvenido a Turespacio',
            'fecha' => date('Y-m-d')
        ];
    }
}
?>

// ajax/controlador.php
<?php
header('Content-Type: application/json');
require_once '../models/Database.php';

$db = new Database();
$accion = $_GET['accion'] ?? '';

switch ($accion) {
    case 'obtenerPosts':
        echo json_encode($db->obtenerPosts());
        break;
    case 'obtenerComentarios':
        $postId = $_GET['post_id'] ?? 0;
        echo json_encode($db->obtenerComentarios($postId));
        break;
    case 'obtenerDatos':
        echo json_encode($db->obtenerDatosGenerales());
        break;
    default:
        echo json_encode(['error' => 'Acción no válida']);
        break;
}
?>
